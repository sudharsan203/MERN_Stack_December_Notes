### Step 2: Understanding Node.js Inbuilt Modules

Node.js comes with many built-in modules that don't require installation.

#### Common Inbuilt Modules

| Module | Description | Common Use Cases |
|--------|-------------|------------------|
| **http** | Create HTTP servers and clients | Building web servers, REST APIs |
| **fs** | File system operations | Read/write files, directory operations |
| **path** | File and directory path utilities | Cross-platform path handling |
| **os** | Operating system information | Get system info, memory, CPU |
| **events** | Event emitter functionality | Custom event handling |
| **stream** | Streaming data | Large file processing |
| **url** | URL parsing and formatting | Parse query strings, URLs |
| **querystring** | Parse URL query strings | Handle form data |
| **crypto** | Cryptographic functionality | Hashing, encryption |
| **util** | Utility functions | Promisify, inspect objects |

## 2. Create Server using HTTP Module

### Step 1: Create Basic HTTP Server

Create file: `http-server.js`

```javascript
const http = require('http');

// Create server
const server = http.createServer((req, res) => {
    // Log request details
    console.log(`${req.method} ${req.url}`);
    
    // Set response headers
    res.writeHead(200, { 
        'Content-Type': 'text/html',
        'Access-Control-Allow-Origin': '*'
    });
    
    // Send response
    res.end('<h1>Hello from Node.js HTTP Server!</h1>');
});

// Start server
const PORT = 3000;
server.listen(PORT, () => {
    console.log(`âœ… HTTP Server running at http://localhost:${PORT}`);
});
```

### Step 2: Create HTTP Server with Routing

Create file: `http-server-routing.js`

```javascript
const http = require('http');
const url = require('url');

const server = http.createServer((req, res) => {
    const parsedUrl = url.parse(req.url, true);
    const pathname = parsedUrl.pathname;
    const method = req.method;
    
    // Set common headers
    res.setHeader('Content-Type', 'application/json');
    res.setHeader('Access-Control-Allow-Origin', '*');
    
    // Routing
    if (pathname === '/' && method === 'GET') {
        res.writeHead(200);
        res.end(JSON.stringify({ message: 'Welcome to Home Page' }));
    } 
    else if (pathname === '/api/users' && method === 'GET') {
        const users = [
            { id: 1, name: 'John Doe', email: 'john@example.com' },
            { id: 2, name: 'Jane Smith', email: 'jane@example.com' }
        ];
        res.writeHead(200);
        res.end(JSON.stringify({ success: true, data: users }));
    } 
    else if (pathname === '/api/products' && method === 'GET') {
        const products = [
            { id: 1, name: 'Laptop', price: 999 },
            { id: 2, name: 'Phone', price: 699 }
        ];
        res.writeHead(200);
        res.end(JSON.stringify({ success: true, data: products }));
    }
    else if (pathname === '/api/users' && method === 'POST') {
        let body = '';
        
        req.on('data', chunk => {
            body += chunk.toString();
        });
        
        req.on('end', () => {
            const newUser = JSON.parse(body);
            res.writeHead(201);
            res.end(JSON.stringify({ 
                success: true, 
                message: 'User created',
                data: newUser 
            }));
        });
    }
    else {
        res.writeHead(404);
        res.end(JSON.stringify({ success: false, message: 'Route not found' }));
    }
});

const PORT = 3000;
server.listen(PORT, () => {
    console.log(`âœ… Server with routing running at http://localhost:${PORT}`);
});
```

### Step 3: Create Express Server with Full CRUD API

Create file: `express-crud-api.js`

```javascript
const express = require('express');
const app = express();

// Middleware
app.use(express.json());
app.use(express.urlencoded({ extended: true }));

// In-memory database (array)
let users = [
    { id: 1, name: 'John Doe', email: 'john@example.com', age: 30 },
    { id: 2, name: 'Jane Smith', email: 'jane@example.com', age: 25 },
    { id: 3, name: 'Bob Johnson', email: 'bob@example.com', age: 35 }
];

// Logging middleware
app.use((req, res, next) => {
    console.log(`${new Date().toISOString()} - ${req.method} ${req.url}`);
    next();
});

// ============================================
// ROUTES
// ============================================

// GET - Home route
app.get('/', (req, res) => {
    res.json({ 
        message: 'Welcome to Express CRUD API',
        endpoints: {
            getUsers: 'GET /api/users',
            getUser: 'GET /api/users/:id',
            createUser: 'POST /api/users',
            updateUser: 'PUT /api/users/:id',
            deleteUser: 'DELETE /api/users/:id'
        }
    });
});

// GET - Get all users
app.get('/api/users', (req, res) => {
    res.json({ 
        success: true, 
        count: users.length,
        data: users 
    });
});

// GET - Get single user by ID
app.get('/api/users/:id', (req, res) => {
    const userId = parseInt(req.params.id);
    const user = users.find(u => u.id === userId);
    
    if (!user) {
        return res.status(404).json({ 
            success: false, 
            message: 'User not found' 
        });
    }
    
    res.json({ success: true, data: user });
});

// POST - Create new user
app.post('/api/users', (req, res) => {
    const { name, email, age } = req.body;
    
    // Validation
    if (!name || !email) {
        return res.status(400).json({ 
            success: false, 
            message: 'Name and email are required' 
        });
    }
    
    // Create new user
    const newUser = {
        id: users.length + 1,
        name,
        email,
        age: age || null
    };
    
    users.push(newUser);
    
    res.status(201).json({ 
        success: true, 
        message: 'User created successfully',
        data: newUser 
    });
});

// PUT - Update user
app.put('/api/users/:id', (req, res) => {
    const userId = parseInt(req.params.id);
    const userIndex = users.findIndex(u => u.id === userId);
    
    if (userIndex === -1) {
        return res.status(404).json({ 
            success: false, 
            message: 'User not found' 
        });
    }
    
    // Update user
    const { name, email, age } = req.body;
    users[userIndex] = {
        ...users[userIndex],
        name: name || users[userIndex].name,
        email: email || users[userIndex].email,
        age: age || users[userIndex].age
    };
    
    res.json({ 
        success: true, 
        message: 'User updated successfully',
        data: users[userIndex] 
    });
});

// DELETE - Delete user
app.delete('/api/users/:id', (req, res) => {
    const userId = parseInt(req.params.id);
    const userIndex = users.findIndex(u => u.id === userId);
    
    if (userIndex === -1) {
        return res.status(404).json({ 
            success: false, 
            message: 'User not found' 
        });
    }
    
    const deletedUser = users.splice(userIndex, 1)[0];
    
    res.json({ 
        success: true, 
        message: 'User deleted successfully',
        data: deletedUser 
    });
});

// 404 Handler
app.use((req, res) => {
    res.status(404).json({ 
        success: false, 
        message: 'Route not found' 
    });
});

// Error handling middleware
app.use((err, req, res, next) => {
    console.error(err.stack);
    res.status(500).json({ 
        success: false, 
        message: 'Internal server error' 
    });
});

// Start server
const PORT = 5000;
app.listen(PORT, () => {
    console.log(`âœ… Express CRUD API running at http://localhost:${PORT}`);
    console.log(`ğŸ“ Try: http://localhost:${PORT}/api/users`);
});
```

```yaml
net:
  port: 27017
  bindIp: 127.0.0.1

security:
  authorization: disabled
```

### Step 4: READ Operations (QUERY)

```javascript
// Find all documents
db.users.find()

// Find all with pretty format
db.users.find().pretty()

// Find with condition
db.users.find({ city: "New York" })

// Find single document
db.users.findOne({ name: "John Doe" })

// Find with multiple conditions (AND)
db.users.find({ city: "New York", age: 30 })

// Find with OR condition
db.users.find({
    $or: [
        { city: "New York" },
        { city: "Los Angeles" }
    ]
})

// Find with comparison operators
db.users.find({ age: { $gt: 30 } })  // Greater than
db.users.find({ age: { $gte: 30 } }) // Greater than or equal
db.users.find({ age: { $lt: 30 } })  // Less than
db.users.find({ age: { $lte: 30 } }) // Less than or equal
db.users.find({ age: { $ne: 30 } })  // Not equal

// Find with IN operator
db.users.find({ city: { $in: ["New York", "Chicago"] } })

// Find with regex (pattern matching)
db.users.find({ name: /john/i }) // Case insensitive

// Projection (select specific fields)
db.users.find({}, { name: 1, email: 1 })      // Include only name and email
db.users.find({}, { age: 0, createdAt: 0 })   // Exclude age and createdAt

// Sorting
db.users.find().sort({ age: 1 })  // Ascending
db.users.find().sort({ age: -1 }) // Descending

// Limit results
db.users.find().limit(2)

// Skip results (pagination)
db.users.find().skip(2).limit(2)

// Count documents
db.users.countDocuments()
db.users.countDocuments({ city: "New York" })
```

### Step 5: UPDATE Operations

```javascript

    { $set: { age: 31, city: "Miami" } }  // Update

    { city: "New York" },
    { $set: { country: "USA" } }
)

// Increment value

    { name: "John Doe" },

)

// Add to array
db.users.updateOne(
    { name: "John Doe" },
    { $push: { hobbies: "swimming" } }
)

// Remove from array
db.users.updateOne(
    { name: "John Doe" },
    { $pull: { hobbies: "swimming" } }
)

// Rename field
db.users.updateOne(
    { name: "John Doe" },
    { $rename: { "age": "userAge" } }
)

// Unset (remove) field
db.users.updateOne(
    { name: "John Doe" },
    { $unset: { city: "" } }
)

// Replace entire document
db.users.replaceOne(
    { name: "John Doe" },

        name: "John Doe",

    }
)

    { name: "New User" },
    { $set: { email: "newuser@example.com", age: 25 } },
    { upsert: true }
)
```

```javascript
import React, { useState } from 'react';
import './TodoList.css';

function TodoList() {
  const [todos, setTodos] = useState([]);
  const [input, setInput] = useState('');

  const addTodo = () => {
    if (input.trim() !== '') {
      setTodos([...todos, { id: Date.now(), text: input, completed: false }]);
      setInput('');
    }
  };

  const toggleTodo = (id) => {
    setTodos(todos.map(todo =>
      todo.id === id ? { ...todo, completed: !todo.completed } : todo
    ));
  };

  const deleteTodo = (id) => {
    setTodos(todos.filter(todo => todo.id !== id));
  };

  return (
    <div className="todo-container">
      <h2>ğŸ“ Todo List</h2>
      <div className="todo-input-group">
        <input
          type="text"
          value={input}
          onChange={(e) => setInput(e.target.value)}
          onKeyPress={(e) => e.key === 'Enter' && addTodo()}
          placeholder="Add a new task..."
          className="todo-input"
        />
        <button onClick={addTodo} className="add-btn">Add</button>
      </div>
      <ul className="todo-list">
        {todos.map(todo => (
          <li key={todo.id} className={`todo-item ${todo.completed ? 'completed' : ''}`}>
            <span onClick={() => toggleTodo(todo.id)} className="todo-text">
              {todo.text}
            </span>
            <button onClick={() => deleteTodo(todo.id)} className="delete-btn">âŒ</button>
          </li>
        ))}
      </ul>
    </div>
  );
}

export default TodoList;
```
