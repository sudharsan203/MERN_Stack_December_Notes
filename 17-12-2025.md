## 1. What is "use strict"?

### Definition
`"use strict"` is a directive introduced in ECMAScript 5 that enables strict mode in JavaScript. It helps catch common coding mistakes and prevents the use of certain problematic features.

### Benefits
- Catches silent errors and throws them instead
- Prevents accidental global variables
- Eliminates `this` coercion
- Disallows duplicate parameter names
- Makes JavaScript code more secure and optimized

### Examples

**Example 1: Without use strict**
```javascript
function normalMode() {
    undeclaredVariable = 10; // No error - creates global variable
    console.log(undeclaredVariable); // 10
}
normalMode();
```

**Example 2: With use strict**
```javascript
"use strict";
function strictMode() {
    undeclaredVariable = 10; // ReferenceError: undeclaredVariable is not defined
    console.log(undeclaredVariable);
}
strictMode();
```

**Example 3: Strict mode in function scope**
```javascript
function myFunction() {
    "use strict";
    x = 10; // Error: x is not defined
}

function normalFunction() {
    y = 20; // Works fine - creates global variable
}
```

**Example 4: Duplicate parameters**
```javascript
// Without strict mode - second param overwrites first
function sum(a, a, c) {
    return a + a + c;
}
console.log(sum(1, 2, 3)); // 7 (2 + 2 + 3)

// With strict mode - throws error
"use strict";
function strictSum(a, a, c) { // SyntaxError: Duplicate parameter name
    return a + a + c;
}
```

## 2. null, undefined, and Not Defined

### Differences

| Concept | Description | Type | Usage |
|---------|-------------|------|-------|
| **undefined** | Variable declared but not assigned a value | Primitive type | Default value for uninitialized variables |
| **null** | Intentional absence of value | Object (typeof returns 'object') | Explicitly assigned to represent "no value" |
| **Not Defined** | Variable not declared at all | ReferenceError | Accessing non-existent variables |

### Examples

**Example 1: undefined**
```javascript
let a;
console.log(a); // undefined
console.log(typeof a); // "undefined"

function test(param) {
    console.log(param); // undefined if not passed
}
test();

let obj = { name: "John" };
console.log(obj.age); // undefined (property doesn't exist)
```

**Example 2: null**
```javascript
let b = null;
console.log(b); // null
console.log(typeof b); // "object" (known JavaScript quirk)

// Intentional use of null
let user = { name: "Alice", age: 25 };
user = null; // Clear the reference

// API response pattern
function getUser(id) {
    if (id === 0) return null; // No user found
    return { id, name: "User" };
}
```

**Example 3: Not Defined**
```javascript
console.log(c); // ReferenceError: c is not defined

"use strict";
function test() {
    nonExistent = 10; // ReferenceError: nonExistent is not defined
}
```

**Example 4: Comparison**
```javascript
console.log(undefined == null);  // true (loose equality)
console.log(undefined === null); // false (strict equality)
console.log(typeof undefined);   // "undefined"
console.log(typeof null);        // "object"

let x;
let y = null;
let z; // not assigned

console.log(x === undefined); // true
console.log(y === null);      // true
console.log(x == y);          // true
console.log(x === y);         // false
```

## 3. Function Scoped vs Block Scoped

### Function Scoped (var)
Variables declared with `var` are function-scoped. They are accessible throughout the entire function, regardless of block boundaries.

### Block Scoped (let & const)
Variables declared with `let` and `const` are block-scoped. They are only accessible within the block `{}` where they are defined.

### Examples

**Example 1: var - Function Scoped**
```javascript
function functionScopedExample() {
    if (true) {
        var x = 10;
        console.log(x); // 10
    }
    console.log(x); // 10 - accessible outside the block
}
functionScopedExample();
// console.log(x); // Error: x is not defined (outside function)
```

**Example 2: let - Block Scoped**
```javascript
function blockScopedExample() {
    if (true) {
        let y = 20;
        console.log(y); // 20
    }
    // console.log(y); // Error: y is not defined (outside block)
}
blockScopedExample();
```

**Example 3: const - Block Scoped**
```javascript
function constExample() {
    if (true) {
        const z = 30;
        console.log(z); // 30
        // z = 40; // Error: Assignment to constant variable
    }
    // console.log(z); // Error: z is not defined
}
constExample();
```

**Example 4: Loop Scoping**
```javascript
// var in loop
for (var i = 0; i < 3; i++) {
    setTimeout(() => console.log(i), 100);
}
// Output: 3, 3, 3 (var is function-scoped)

// let in loop
for (let j = 0; j < 3; j++) {
    setTimeout(() => console.log(j), 100);
}
// Output: 0, 1, 2 (let is block-scoped)
```

**Example 5: Hoisting Differences**
```javascript
function hoistingExample() {
    console.log(varVariable); // undefined (hoisted)
    // console.log(letVariable); // ReferenceError: Cannot access before initialization
    
    var varVariable = "I'm var";
    let letVariable = "I'm let";
    
    console.log(varVariable); // "I'm var"
    console.log(letVariable); // "I'm let"
}
hoistingExample();
```

**Example 6: Real-world Scenario**
```javascript
function createButtons() {
    // Problem with var
    for (var i = 1; i <= 3; i++) {
        document.body.innerHTML += `<button onclick="alert(${i})">Button ${i}</button>`;
    }
    // All buttons alert 4
    
    // Solution with let
    for (let j = 1; j <= 3; j++) {
        document.body.innerHTML += `<button onclick="alert(${j})">Button ${j}</button>`;
    }
    // Each button alerts its own number
}
```

## 4. Prerequisites to Install Node.js, React.js, and MongoDB

### System Requirements

#### For Node.js
- **Operating System:** Windows 10/11, macOS 10.13+, or Linux
- **RAM:** Minimum 4GB (8GB recommended)
- **Disk Space:** At least 1GB free space
- **Administrator Rights:** Required for installation

#### For React.js
- **Node.js:** Version 14.0 or higher
- **npm:** Version 6.0 or higher (comes with Node.js)
- **Code Editor:** VS Code, Sublime Text, or any text editor
- **Web Browser:** Chrome, Firefox, Edge, or Safari (latest versions)

#### For MongoDB
- **Operating System:** Windows 10/11 (64-bit), macOS 10.13+, or Linux
- **RAM:** Minimum 4GB (8GB+ recommended for production)
- **Disk Space:** Minimum 5GB free space
- **Processor:** 64-bit processor
- **Administrator Rights:** Required for installation

### Software Prerequisites
1. **Text Editor/IDE:** Visual Studio Code (recommended)
2. **Web Browser:** Google Chrome (with React DevTools extension)
3. **Command Line:** PowerShell (Windows), Terminal (macOS/Linux)
4. **Git:** For version control (optional but recommended)

## 5. Installation Steps & Verification

### A. Installing Node.js

#### Installation Steps (Windows)

1. **Download Node.js**
   - Visit: https://nodejs.org/
   - Download LTS (Long Term Support) version
   - Choose Windows Installer (.msi) 64-bit

2. **Run Installer**
   - Double-click the downloaded file
   - Click "Next" on the welcome screen
   - Accept the license agreement
   - Choose installation location (default: C:\Program Files\nodejs)
   - Keep all default features selected
   - Check "Automatically install necessary tools" (optional)
   - Click "Install"

3. **Complete Installation**
   - Click "Finish" when installation completes
   - Restart your computer (recommended)

#### Verification Steps

```powershell
# Check Node.js version
node --version
# Expected output: v20.x.x or v18.x.x

# Check npm version
npm --version
# Expected output: 10.x.x or 9.x.x

# Test Node.js REPL
node
# Type: console.log("Hello Node!")
# Press Ctrl+C twice to exit
```

#### Installation Steps (macOS)

1. Using Homebrew (recommended):
```bash
# Install Homebrew if not installed
/bin/bash -c "$(curl -fsSL https://raw.githubusercontent.com/Homebrew/install/HEAD/install.sh)"

# Install Node.js
brew install node

# Verify installation
node --version
npm --version
```

2. Using official installer:
   - Download from https://nodejs.org/
   - Run the .pkg file
   - Follow installation wizard

### B. Installing React.js

#### Installation Steps

React doesn't have a separate installer. It's installed via npm (Node Package Manager).

**Method 1: Create React App (CRA)**

```powershell
# Create a new React application
npx create-react-app my-app

# Navigate to the project folder
cd my-app

# Start the development server
npm start
```

**Method 2: Using Vite (Faster alternative)**

```powershell
# Create a new React app with Vite
npm create vite@latest my-react-app -- --template react

# Navigate to the project folder
cd my-react-app

# Install dependencies
npm install

# Start the development server
npm run dev
```

#### Verification Steps

```powershell
# After running npm start or npm run dev

# http://localhost:3000 (Create React App)
# or
# http://localhost:5173 (Vite)

```

### C. Installing MongoDB

#### Installation Steps (Windows)

1. **Download MongoDB**
   - Visit: https://www.mongodb.com/try/download/community
   - Select: Windows x64
   - Download the MSI installer

2. **Run Installer**
   - Double-click the .msi file
   - Click "Next" on the welcome screen
   - Accept the license agreement
   - Choose "Complete" installation
   - Install MongoDB as a Service (recommended)
   - Keep default settings:
     - Service Name: MongoDB
     - Data Directory: C:\Program Files\MongoDB\Server\7.0\data\
     - Log Directory: C:\Program Files\MongoDB\Server\7.0\log\
   - Install MongoDB Compass (GUI tool) - recommended
   - Click "Install"

3. **Set Environment Variable**
```powershell
# Add MongoDB bin to PATH
# Go to System Properties > Environment Variables
# Edit Path variable and add:
# C:\Program Files\MongoDB\Server\7.0\bin
```

#### Installation Steps (macOS)

```bash
# Using Homebrew
brew tap mongodb/brew
brew install mongodb-community

# Start MongoDB
brew services start mongodb-community

# Stop MongoDB (when needed)
brew services stop mongodb-community
```

#### Verification Steps

```powershell
# Check MongoDB version
mongod --version
# Expected output: db version v7.x.x

# Start MongoDB Shell
mongosh
# or
mongo

# In the MongoDB shell, test commands:
show dbs
# Should show: admin, config, local databases

# Create a test database
use testdb

# Insert a test document
db.users.insertOne({ name: "Test User", age: 25 })

# Query the document
db.users.find()

# Exit MongoDB shell
exit
```

**Testing MongoDB Connection with Node.js:**

```javascript
// test-mongo.js
const { MongoClient } = require('mongodb');

const uri = "mongodb://localhost:27017";
const client = new MongoClient(uri);

async function testConnection() {
    try {
        await client.connect();
        console.log("✅ Connected to MongoDB successfully!");
        
        const db = client.db("testdb");
        const collections = await db.listCollections().toArray();
        console.log("Collections:", collections);
    } catch (error) {
        console.error("❌ Connection failed:", error);
    } finally {
        await client.close();
    }
}

testConnection();
```

Run the test:
```powershell
# Install MongoDB driver
npm install mongodb

# Run the test file
node test-mongo.js
```

## Common Issues & Solutions

### Node.js Issues
**Issue:** "node is not recognized as internal or external command"
**Solution:** Add Node.js to PATH environment variable or restart terminal

### React Issues
**Issue:** Port 3000 is already in use
**Solution:** Kill the process or use a different port: `PORT=3001 npm start`

### MongoDB Issues
**Issue:** "MongoNetworkError: failed to connect"
**Solution:** Ensure MongoDB service is running: `net start MongoDB`
