## ğŸ”‘ Keywords & Variable Declaration

JavaScript provides three ways to declare variables:

### `var` (Function-scoped)
```javascript
var name = "MERN";
var age = 25;
// Can be re-declared and updated
// Function-scoped (not block-scoped)
```

### `let` (Block-scoped)
```javascript
let number = 5;
let city = "New York";
// Cannot be re-declared in same scope
// Can be updated
// Block-scoped
```

### `const` (Block-scoped, Immutable reference)
```javascript
const id = 10;
const PI = 3.14159;
// Cannot be re-declared or updated
// Must be initialized at declaration
// Block-scoped
```

### Variable Declaration Best Practices
```javascript
// âœ… Good Practice
const MAX_USERS = 100;        // Use const for values that won't change
let currentCount = 0;         // Use let for values that will change
const config = { api: "..." }; // Objects/Arrays with const (reference is constant)

// âŒ Avoid
var globalVar = "something";  // Avoid var, use let/const instead
```

## ğŸ“Š Data Types

JavaScript has 8 fundamental data types:

### 1. String
```javascript
let text = "MERN Stack";
let singleQuote = 'Hello World';
let template = `My name is ${text}`;
console.log(text);        // "MERN Stack"
console.log(text.length); // 10
```

### 2. Number
```javascript
let integer = 12;
let decimal = 3.14;
let negative = -42;
let infinity = Infinity;
console.log(typeof integer); // "number"
```

### 3. Boolean
```javascript
let isActive = true;
let isDeleted = false;
console.log(typeof isActive); // "boolean"
```

### 4. Object
```javascript
let person = {
    name: "MEAN",
    age: 25,
    city: "New York",
    isEmployed: true
};
console.log(person.name);      // "MEAN"
console.log(person["age"]);    // 25
```

### 5. Array
```javascript
let skills = ["HTML", "CSS", "JavaScript"];
let mixed = [
    "MERN",
    2,
    true,
    {
        name: "js",
        exp: "1 year"
    }
];
console.log(skills[0]);        // "HTML"
console.log(skills.length);    // 3
```

### 6. Null
```javascript
let emptyValue = null;
console.log(emptyValue);       // null
console.log(typeof null);      // "object" (JavaScript quirk)
```

### 7. Undefined
```javascript
let notAssigned;
console.log(notAssigned);      // undefined
console.log(typeof undefined); // "undefined"
```

### 8. Symbol (ES6)
```javascript
let sym1 = Symbol('description');
let sym2 = Symbol('description');
console.log(sym1 === sym2);    // false (unique)
```

## âš™ï¸ Functions

Functions are reusable blocks of code.

### Function Declaration
```javascript
function sum(a, b) {
    console.log("Function called");
    return a + b;
}

let result = sum(5, 10);
console.log(result); // 15
```

### Function with Parameters
```javascript
function greet(name) {
    return `Hello, ${name}!`;
}

console.log(greet("Alice")); // "Hello, Alice!"
```

### Arrow Functions (ES6)
```javascript
// Basic arrow function
const mult = () => {
    console.log("Arrow Function");
};
mult();

// Arrow function with parameters
const multiply = (a, b) => {
    return a * b;
};
console.log(multiply(4, 5)); // 20

// Shorter syntax (implicit return)
const add = (a, b) => a + b;
console.log(add(3, 7)); // 10

// Single parameter (no parentheses needed)
const square = x => x * x;
console.log(square(4)); // 16
```

### Function Expression
```javascript
const divide = function(a, b) {
    return a / b;
};
console.log(divide(10, 2)); // 5
```

### Default Parameters
```javascript
function greetUser(name = "Guest") {
    return `Welcome, ${name}!`;
}
console.log(greetUser());        // "Welcome, Guest!"
console.log(greetUser("John"));  // "Welcome, John!"
```

## ğŸ”€ Control Flow

### Conditional Statements

#### if-else Statement
```javascript
let age = 18;

if (age >= 18) {
    console.log("Adult");
} else if (age >= 13) {
    console.log("Teenager");
} else {
    console.log("Child");
}
```

#### switch Statement
```javascript
let day = "Monday";

switch(day) {
    case "Monday":
        console.log("Start of the week");
        break;
    case "Friday":
        console.log("End of work week");
        break;
    case "Saturday":
    case "Sunday":
        console.log("Weekend!");
        break;
    default:
        console.log("Midweek");
}
```

#### Ternary Operator
```javascript
let age = 20;
let status = age >= 18 ? "Adult" : "Minor";
console.log(status); // "Adult"
```

### Loops

#### for Loop
```javascript
for (let i = 0; i < 5; i++) {
    console.log(`Iteration ${i}`);
}
// Output: Iteration 0, 1, 2, 3, 4
```

#### while Loop
```javascript
let count = 0;
while (count < 3) {
    console.log(`Count: ${count}`);
    count++;
}
```

#### do-while Loop
```javascript
let num = 0;
do {
    console.log(num);
    num++;
} while (num < 3);
```

#### for...of Loop (Arrays)
```javascript
let fruits = ["Apple", "Banana", "Orange"];
for (let fruit of fruits) {
    console.log(fruit);
}
```

#### for...in Loop (Objects)
```javascript
let person = { name: "John", age: 30, city: "NYC" };
for (let key in person) {
    console.log(`${key}: ${person[key]}`);
}
```

## ğŸ”„ CRUD Operations

**CRUD** = Create, Read, Update, Delete

### Array CRUD Operations
```javascript
let users = [];

// CREATE - Add new items
users.push({ id: 1, name: "Alice" });
users.push({ id: 2, name: "Bob" });
console.log(users);

// READ - Access items
console.log(users[0]);           // First user
console.log(users.length);       // Number of users

// UPDATE - Modify items
users[0].name = "Alice Smith";
console.log(users[0]);

// DELETE - Remove items
users.pop();                     // Remove last item
users.splice(0, 1);              // Remove item at index 0
console.log(users);
```

### Object CRUD Operations
```javascript
let user = {};

// CREATE - Add properties
user.id = 1;
user.name = "John";
user.email = "john@example.com";
console.log(user);

// READ - Access properties
console.log(user.name);
console.log(user["email"]);

// UPDATE - Modify properties
user.name = "John Doe";
user.age = 30;
console.log(user);

// DELETE - Remove properties
delete user.age;
console.log(user);
```

## ğŸ”¤ String Operations

### Common String Methods
```javascript
let text = "MERN Stack Developer";

// Length
console.log(text.length);              // 20

// Case conversion
console.log(text.toLowerCase());       // "mern stack developer"
console.log(text.toUpperCase());       // "MERN STACK DEVELOPER"

// Substring
console.log(text.substring(0, 4));     // "MERN"
console.log(text.slice(5, 10));        // "Stack"

// Search
console.log(text.indexOf("Stack"));    // 5
console.log(text.includes("MERN"));    // true

// Replace
console.log(text.replace("MERN", "MEAN")); // "MEAN Stack Developer"

// Split
console.log(text.split(" "));          // ["MERN", "Stack", "Developer"]

// Trim whitespace
let padded = "  Hello  ";
console.log(padded.trim());            // "Hello"

// Repeat
console.log("M".repeat(25));           // "MMMMMMMMMMMMMMMMMMMMMMMMM"

// Template literals
let name = "Alice";
let greeting = `Hello, ${name}!`;
console.log(greeting);                 // "Hello, Alice!"
```

### String Concatenation
```javascript
let firstName = "John";
let lastName = "Doe";

// Using + operator
let fullName1 = firstName + " " + lastName;

// Using template literals (preferred)
let fullName2 = `${firstName} ${lastName}`;

console.log(fullName1); // "John Doe"
console.log(fullName2); // "John Doe"
```

## ğŸ“¦ Object Operations

### Creating Objects
```javascript
// Object literal
let person = {
    name: "JavaScript",
    born: 1995,
    age: 30,
    skills: ["Web", "Mobile", "Desktop"]
};

// Using new Object()
let car = new Object();
car.brand = "Tesla";
car.model = "Model 3";
```

### Accessing Object Properties
```javascript
let person = {
    name: "JS",
    born: 1900,
    age: 35,
    city: "Web"
};

// Dot notation
console.log(person.name);         // "JS"

// Bracket notation
console.log(person["born"]);      // 1900

// Dynamic property access
let prop = "age";
console.log(person[prop]);        // 35
```

### Modifying Objects
```javascript
let person = { name: "Alice", age: 25 };

// Update existing property
person.age = 26;

// Add new property
person.city = "New York";
person["country"] = "USA";

// Delete property
delete person.country;

console.log(person);
// { name: "Alice", age: 26, city: "New York" }
```

### Object Methods
```javascript
let user = {
    firstName: "John",
    lastName: "Doe",
    fullName: function() {
        return `${this.firstName} ${this.lastName}`;
    },
    // ES6 shorthand
    greet() {
        return `Hello, I'm ${this.fullName()}`;
    }
};

console.log(user.fullName());  // "John Doe"
console.log(user.greet());     // "Hello, I'm John Doe"
```

### Object Utility Methods
```javascript
let person = {
    name: "Alice",
    age: 30,
    city: "NYC"
};

// Get all keys
console.log(Object.keys(person));
// ["name", "age", "city"]

// Get all values
console.log(Object.values(person));
// ["Alice", 30, "NYC"]

// Get entries (key-value pairs)
console.log(Object.entries(person));
// [["name", "Alice"], ["age", 30], ["city", "NYC"]]

// Check if property exists
console.log(person.hasOwnProperty("name")); // true

// Copy/merge objects
let clone = Object.assign({}, person);
let merged = Object.assign({}, person, { country: "USA" });
```

## ğŸ“‹ Array Operations

### Creating Arrays
```javascript
// Array literal
let fruits = ["Apple", "Banana", "Orange"];

// Using Array constructor
let numbers = new Array(1, 2, 3, 4, 5);

// Empty array with length
let empty = new Array(5); // [undefined x 5]
```

### Array Methods

#### Adding/Removing Elements
```javascript
let arr = [1, 2, 3];

// Add to end
arr.push(4);              // [1, 2, 3, 4]

// Add to beginning
arr.unshift(0);           // [0, 1, 2, 3, 4]

// Remove from end
arr.pop();                // [0, 1, 2, 3]

// Remove from beginning
arr.shift();              // [1, 2, 3]

// Add/remove at specific index
arr.splice(1, 1, 5, 6);   // [1, 5, 6, 3] (removes 1 item at index 1, adds 5,6)
```

#### Searching Arrays
```javascript
let fruits = ["Apple", "Banana", "Orange", "Banana"];

// Find index
console.log(fruits.indexOf("Banana"));     // 1

// Find last index
console.log(fruits.lastIndexOf("Banana")); // 3

// Check if exists
console.log(fruits.includes("Apple"));     // true

// Find element
let result = fruits.find(f => f.length > 5);
console.log(result);                       // "Banana"

// Find index with condition
let index = fruits.findIndex(f => f === "Orange");
console.log(index);                        // 2
```

#### Transforming Arrays
```javascript
let numbers = [1, 2, 3, 4, 5];

// Map - transform each element
let doubled = numbers.map(n => n * 2);
console.log(doubled);     // [2, 4, 6, 8, 10]

// Filter - select elements
let evens = numbers.filter(n => n % 2 === 0);
console.log(evens);       // [2, 4]

// Reduce - combine elements
let sum = numbers.reduce((acc, n) => acc + n, 0);
console.log(sum);         // 15

// ForEach - iterate
numbers.forEach(n => console.log(n * 2));

// Sort
let unsorted = [3, 1, 4, 1, 5];
unsorted.sort((a, b) => a - b);
console.log(unsorted);    // [1, 1, 3, 4, 5]

// Reverse
console.log(unsorted.reverse()); // [5, 4, 3, 1, 1]
```

#### Array Utilities
```javascript
let arr1 = [1, 2, 3];
let arr2 = [4, 5, 6];

// Concatenate
let combined = arr1.concat(arr2);
console.log(combined);    // [1, 2, 3, 4, 5, 6]

// Spread operator (ES6)
let merged = [...arr1, ...arr2];
console.log(merged);      // [1, 2, 3, 4, 5, 6]

// Slice - extract portion
let slice = combined.slice(2, 5);
console.log(slice);       // [3, 4, 5]

// Join - to string
console.log(arr1.join("-")); // "1-2-3"

// Check if array
console.log(Array.isArray(arr1)); // true
```

#### Multi-dimensional Arrays
```javascript
let matrix = [
    [1, 2, 3],
    [4, 5, 6],
    [7, 8, 9]
];

console.log(matrix[0][0]);    // 1
console.log(matrix[1][2]);    // 6

// Iterate 2D array
matrix.forEach(row => {
    row.forEach(num => {
        console.log(num);
    });
});
```

## ğŸ¯ Practice Examples

### Example 1: User Management
```javascript
let users = [
    { id: 1, name: "Alice", age: 25 },
    { id: 2, name: "Bob", age: 30 },
    { id: 3, name: "Charlie", age: 35 }
];

// READ - Get all users
console.log(users);

// READ - Find specific user
let user = users.find(u => u.id === 2);
console.log(user);

// UPDATE - Update user
let userIndex = users.findIndex(u => u.id === 2);
users[userIndex].age = 31;

// DELETE - Remove user
users = users.filter(u => u.id !== 3);

console.log(users);
```

### Example 2: String Manipulation
```javascript
let email = "  user@EXAMPLE.com  ";

// Clean and format
let cleaned = email.trim().toLowerCase();
console.log(cleaned); // "user@example.com"

// Validate
let isValid = cleaned.includes("@") && cleaned.includes(".");
console.log(isValid); // true

// Extract parts
let parts = cleaned.split("@");
console.log(parts[0]); // "user"
console.log(parts[1]); // "example.com"
```

### Example 3: Data Processing
```javascript
let products = [
    { name: "Laptop", price: 1000 },
    { name: "Mouse", price: 25 },
    { name: "Keyboard", price: 75 }
];

// Calculate total
let total = products.reduce((sum, p) => sum + p.price, 0);
console.log(`Total: $${total}`); // "Total: $1100"

// Get expensive items
let expensive = products.filter(p => p.price > 50);
console.log(expensive);

// Add tax
let withTax = products.map(p => ({
    ...p,
    priceWithTax: p.price * 1.1
}));
console.log(withTax);
```
